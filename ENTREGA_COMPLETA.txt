🚀 COMO COMEÇAR

Passo 1: Setup (2 minutos)
  cd alici-core
  python -m venv venv
  source venv/bin/activate
  pip install -r ../requirements.txt
  cp .env.example .env
  # Edite .env com DATABASE_URL

Passo 2: Validar (1 minuto)
  python teste_completo.py
  # Output: ✅ TUDO FUNCIONANDO!

Passo 3: Demo (2 minutos)
  python demo.py
  # Vê toda a arquitetura funcionando

Passo 4: API (30 segundos)
  cd api
  uvicorn main:app --reload
  # Acesse: http://localhost:8000/docs

Passo 5: Treinar (5 minutos)
  python training/trainer.py
  # Treina 5 épocas com dados dummy

Passo 6: Deploy (10 minutos)
  git push
  # Configure DATABASE_URL em Render
  # Deploy automático
